-- –Ґ–∞–±–ї–Є—Ж–∞ –њ–ї–∞–љ–Њ–≤
CREATE TABLE FLOORPLAN (
    ID_FLOORPLAN INTEGER NOT NULL,
    NAME VARCHAR(100) NOT NULL,
    DESCRIPTION VARCHAR(255),
    IMAGE VARCHAR(255) NOT NULL,
    WIDTH INTEGER DEFAULT 800,
    HEIGHT INTEGER DEFAULT 600,
    CREATED TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- –Ґ–∞–±–ї–Є—Ж–∞ —В–Њ—З–µ–Ї –љ–∞ –њ–ї–∞–љ–µ (–Є—Б–њ—А–∞–≤–ї–µ–љ–Њ: TYPE -> POINT_TYPE)
CREATE TABLE FLOORPLAN_POINT (
    ID_POINT INTEGER NOT NULL,
    ID_FLOORPLAN INTEGER NOT NULL,
    X_POS FLOAT DEFAULT 0,
    Y_POS FLOAT DEFAULT 0,
    ID_DEV INTEGER,
    POINT_TYPE VARCHAR(20) DEFAULT 'door',
    LABEL VARCHAR(100)
);

-- –У–µ–љ–µ—А–∞—В–Њ—А—Л –і–ї—П Firebird
CREATE GENERATOR GEN_FLOORPLAN_ID;
CREATE GENERATOR GEN_FLOORPLAN_POINT_ID;

-- –Ґ—А–Є–≥–≥–µ—А—Л –і–ї—П –∞–≤—В–Њ–Є–љ–Ї—А–µ–Љ–µ–љ—В–∞
SET TERM ^ ;
CREATE TRIGGER TRG_FLOORPLAN_BI FOR FLOORPLAN
ACTIVE BEFORE INSERT POSITION 0
AS
BEGIN
    IF (NEW.ID_FLOORPLAN IS NULL) THEN
        NEW.ID_FLOORPLAN = GEN_ID(GEN_FLOORPLAN_ID, 1);
END^

CREATE TRIGGER TRG_FLOORPLAN_POINT_BI FOR FLOORPLAN_POINT
ACTIVE BEFORE INSERT POSITION 0
AS
BEGIN
    IF (NEW.ID_POINT IS NULL) THEN
        NEW.ID_POINT = GEN_ID(GEN_FLOORPLAN_POINT_ID, 1);
END^
SET TERM ; ^

-- –Т–љ–µ—И–љ–Є–є –Ї–ї—О—З
ALTER TABLE FLOORPLAN_POINT ADD CONSTRAINT FK_FLOORPLAN_POINT
    FOREIGN KEY (ID_FLOORPLAN) REFERENCES FLOORPLAN(ID_FLOORPLAN) ON DELETE CASCADE;
